# demo_compare.txt - 用于比较三种内存分配策略的workload
# 设计突出first fit, best fit, worst fit的差异
# first fit: 保留高地址大分区，有利于大进程；低地址碎片多
# best fit: 产生最多碎片（小块多）
# worst fit: 不利于大进程（大块被分割）

# 初始分配中等块
alloc 1 100
alloc 2 150
alloc 3 200
alloc 4 120

# 释放部分，创建碎片
free 2  # 释放150块，创建碎片
free 4  # 释放120块

# 现在分配中等块，观察策略选择
alloc 5 80   # first fit选第一个合适(150-350), best fit选最小合适, worst fit选最大
alloc 6 60   # 继续分配

# 释放更多，增加碎片复杂性
free 1
free 3

# 分配多个小块，增加碎片复杂性（best fit会产生更多小碎片）
alloc 7 30
alloc 8 40
alloc 9 25
alloc 10 35
alloc 13 20
alloc 14 45

# 释放一些，创建更多小碎片
free 5
free 7
free 9
free 13

# 现在尝试分配大块，（体现first保留大分区，worst分割大块）
alloc 15 250  # 大块分配

# 最后分配中等块，观察碎片影响
alloc 16 100

# 显示最终分布
show

# 显示统计
stats
